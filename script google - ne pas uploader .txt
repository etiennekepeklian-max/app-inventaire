function hashPassword(password) {
  var digest = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, password); var hashStr = '';
  for (var i = 0; i < digest.length; i++) { var byte = digest[i]; if (byte < 0) byte += 256; var byteStr = byte.toString(16); if (byteStr.length == 1) byteStr = '0' + byteStr; hashStr += byteStr; }
  return hashStr;
}

function verifierToken(login, token, idBase) {
  if (!login || !token) return false;
  var sheetUser = SpreadsheetApp.openById(idBase).getSheetByName("user"); var data = sheetUser.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] === login && data[i][2] === token && new Date() < new Date(data[i][3])) return data[i][4] || "standard";
  }
  return false;
}
